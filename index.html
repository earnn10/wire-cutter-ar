// --- [คัดลอกโค้ดนี้ไปวางทับฟังก์ชันเดิม] --- // ตัวแปรสำหรับเก็บค่าความสว่างเริ่มต้น (Baseline) let initialBrightness = { red: 255, yellow: 255, blue: 255 }; let isFirstScan = true; // สำหรับกำหนดค่าเริ่มต้น function checkWireObstruction(colorName, sensitivityThreshold) { const wire = document.getElementById(colorName + '-wire'); if (!wire) return; const rect = wire.getBoundingClientRect(); const containerRect = document.getElementById('game-canvas-container').getBoundingClientRect(); // คำนวณพิกเซลสัมพัทธ์ในกรอบวิดีโอ const x = rect.left - containerRect.left; const y = rect.top - containerRect.top; const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); canvas.width = 10; // สแกนพื้นที่เล็กๆ พอ canvas.height = 10; try { // ดึงภาพจากวิดีโอเฉพาะจุดที่สายไฟเส้นนั้นตั้งอยู่ ctx.drawImage(videoElement, x, y, rect.width, rect.height, 0, 0, canvas.width, canvas.height); const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height); const data = imageData.data; let totalBrightness = 0; for (let i = 0; i < data.length; i += 4) { // คำนวณความสว่าง (Brightness) ของพิกเซล (สูตร Luminance) let brightness = (data[i] * 0.299 + data[i+1] * 0.587 + data[i+2] * 0.114); totalBrightness += brightness; } let avgBrightness = totalBrightness / (data.length / 4); // --- ส่วนที่แก้ไข: ระบบตรวจจับการเปลี่ยนแปลงของแสง (Advanced Sensitivity) --- // 1. ถ้าเป็นการสแกนครั้งแรกเมื่อเริ่มเกม ให้เก็บค่าความสว่างปกติไว้เป็นฐาน (Baseline) if (isFirstScan) { initialBrightness[colorName] = avgBrightness; if (colorName === 'blue') isFirstScan = false; // เมื่อสแกนครบ 3 เส้น ปิดสแกนครั้งแรก return; // ยังไม่ต้องตรวจสอบ } // 2. คำนวณเปอร์เซ็นต์ความสว่างที่ลดลงจากค่าเริ่มต้น // (ปกติจะอยู่ระหว่าง 0 - 100, ถ้ามืดลง ค่าจะเข้าใกล้ 100) const percentDrop = ((initialBrightness[colorName] - avgBrightness) / initialBrightness[colorName]) * 100; // 3. ปรับค่าความไว (Sensitivity Threshold): // ปกติค่า percentDrop จะอยู่ราวๆ 20-40% เมื่อโดนมือบัง // ผมจะตั้งค่าเริ่มต้นที่ 25 (คือถ้ามืดลง 25% ก็นับว่าตัดสำเร็จ) const sensitivity = 25; // 4. เงื่อนไขชนะ: ถ้าความสว่างลดลงเกินค่าความไวที่ตั้งไว้ if (percentDrop > sensitivity) { triggerWireCut(colorName); } } catch (e) { console.log("สแกนภาพผิดพลาดชั่วคราว"); } } // และต้องเพิ่มคำสั่งรีเซ็ตค่า baseline ในฟังก์ชัน resetGame() ด้วยครับ // [คัดลอกส่วนนี้ไปวางในฟังก์ชัน resetGame() ที่มีอยู่แล้ว] function resetGame() { // ...
